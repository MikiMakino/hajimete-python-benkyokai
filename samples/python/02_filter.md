# サンプル：条件で絞り込む

```python
# カテゴリの種類を確認する
print(整理済み["カテゴリ"].unique())

# 文字条件で絞り込む
整理済み[整理済み["カテゴリ"] == "文具"]

# 数値条件で絞り込む
整理済み[整理済み["金額"] >= 2000]

# 2つの条件を組み合わせる（AND）
整理済み[(整理済み["カテゴリ"] == "文具") & (整理済み["金額"] >= 2000)]

# マスタ未登録のデータを確認する
未登録 = 整理済み[整理済み["商品名"].isna()]
print(f"全体 {len(整理済み)} 件中、マスタ未登録は {len(未登録)} 件です。")
```
